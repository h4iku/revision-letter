\usepackage{chngcntr}

% Environment for review comments
\newcounter{revcomcounter}[section]
\renewcommand{\therevcomcounter}{\arabic{section}.\arabic{revcomcounter}}

% Review comment environment
\newenvironment{revcom}[1][]{%
    \refstepcounter{revcomcounter}%
    \ifx\\#1\\%
    \else
        \label{#1}%
    \fi
    \begin{trivlist}\item\textbf{Comment~\therevcomcounter: }\em%
        }%
        {\end{trivlist}}%

% Environment for responding to review comments
\newenvironment{response}[1][]{%
    \begin{trivlist}\item\textbf{Response: }%
        }%
        {\end{trivlist}}%

% Reset comment counter in each section
\counterwithin*{revcomcounter}{section}
